import { startLanguageServer } from 'dygram';

startLanguageServer({
  port: 3000
});

PROGRAMMATIC API

Use DyGram as a library in your Node.js applications:

import { MachineExecutor } from 'dygram';

const machineData = {
  title: "Test Machine",
  nodes: [
    { name: "start", type: "state" },
    { name: "process", type: "Task", attributes: [
      { name: "prompt", value: "Process data" }
    ]},
    { name: "end", type: "state" }
  ],
  edges: [
    { source: "start", target: "process" },
    { source: "process", target: "end" }
  ]
};

const executor = await MachineExecutor.create(machineData);
const result = await executor.execute();

WEB EMBEDDING

Embed the DyGram editor in your web applications:

<script src="dygram-web.js"></script>
<div id="editor"></div>
<script>
  DyGram.createEditor({
    container: '#editor',
    value: 'machine "Hello"'
  });
</script>

CI/CD INTEGRATION

Run DyGram in your continuous integration pipelines:

# GitHub Actions example
- name: Validate DyGram files
  run: |
    npm install -g dygram
    dygram validate **/*.mach

EXPORT FORMATS

Generate diagrams and documentation:

# Generate Mermaid diagram
dygram export --format mermaid file.mach

# Export to JSON
dygram export --format json file.mach
